C\# .NETмқҖ Managed CodeмқҙлӢӨ.  .NETмқҳ кө¬м„ұмҡ”мҶҢмқё CLR(Common Language Runtime)м—җм„ң мӢӨн–үлҗҳлҠ” мҪ”л“ң мң нҳ•мқҙлӢӨ. мғҒлҢҖм ҒмңјлЎң Unmanaged CodeлҠ” CLRмқҳ к°ңмһ… м—Ҷмқҙ мҡҙмҳҒмІҙм ңлӮҳ н•ҳл“ңмӣЁм–ҙм—җм„ң м§Ғм ‘ мӢӨн–үлҗҳлҠ” мҪ”л“ң мң нҳ•мқҙлӢӨ. C\#ліҙлӢӨ лЁёмӢ  мҲҳмӨҖм—җ лҚ” к°Җк№Ңмҡҙ C, C\++ мҷҖ к°ҷмқҖ м–ём–ҙлЎң мһ‘м„ұлҗңлӢӨ.<sup>1</sup> Managed Codeмқё C\#мқҖ лҹ°нғҖмһ„ нҷҳкІҪм—җм„ң мһҗлҸҷмңјлЎң л©”лӘЁлҰ¬ кҙҖлҰ¬(GC, Garbage Collection)лҘј кҙҖлҰ¬н•ңлӢӨ.

IDisposableмқ„ кө¬нҳ„н•ҳлҠ” к°ңмІҙлҘј мӮ¬мҡ©н•  л•Ң `using StreamReader`мҷҖ к°ҷмқҙ мӮ¬мҡ©н•ҳл©ҙ н•ЁмҲҳмқҳ лё”лЎқмқҙ лҒқлӮ  л•Ң мһҗлҸҷмңјлЎң л©”лӘЁлҰ¬м—җм„ң к°ңмІҙлҘј л©”лӘЁлҰ¬м—җм„ң н•ҙм ңн•ңлӢӨ.<sup>2</sup> к°„нҳ№ к°ңл°ңмһҗ мӨ‘м—җ м§Ғм ‘ л©”лӘЁлҰ¬лҘј н•ҙм ңн•ҳкі мһҗ н•  л•Ң м•„лһҳмҷҖ к°ҷмқҙ мҪ”л“ңлҘј мһ‘м„ұн•ҳкіӨ н•ҳлҠ” лҚ° мқҙлҹ¬н•ң л°©лІ•мқҖ GCлҘј мҳӨн•ҙн•ң к№ҢлӢӯм—җм„ң лӮҳмҳЁ кІ°кіјмқҙлӢӨ.<sup>3</sup>

###### к°ңм„  м „ мҪ”л“ң

```cs
public static void ClearMemory<T>(this List<T> list)
{
    int id = GC.GetGeneration(list);
    list.Clear();
    GC.Collect(id, GCCollectionMode.Forced);
}
```

`GC.Collect()`лҘј нҳём¶ңн•ҳл©ҙ к°Җл№„м§Җ м»¬л үм…ҳмқҙ мӢӨн–үлҗ  к°ҖлҠҘм„ұмқ„ лҶ’мқҙм§Җл§Ң Unmanaged м–ём–ҙмҷҖ к°ҷмқҙ л°”лЎң л©”лӘЁлҰ¬к°Җ л°ҳнҷҳлҗҳлҠ” кІғмқ„ ліҙмһҘн•ҳм§Җ м•ҠлҠ”лӢӨ. мҰү, нһҢнҠёлҘј мӨҖ м •лҸ„мқҳ м—ӯн• л§Ң н•  лҝҗмқҙкі  нҠ№нһҲ GCлҘј мһҗмЈј нҳём¶ңн•ҳл©ҙ CPU мӢңк°„мқ„ мҶҢлӘЁн•ҳлҠ” мһ‘м—…мқҙкё° л•Ңл¬ём—җ мҳӨнһҲл Ө м„ұлҠҘм—җ м•…мҳҒн–Ҙмқ„ лҜём№  мҲҳ мһҲлӢӨ. 

м•„лһҳмқҳ мҳҲм ңлҠ” к¶ҢмһҘн•ҳлҠ” мҪ”л“ңмҷҖ н•Ёк»ҳ кіөл¶Җ м°Ёмӣҗм—җм„ң м§Ғм ‘ мһ‘м„ұн•ң лӘҮ к°Җм§Җ нҒҙлһҳмҠӨмқҙлӢӨ.





###### к¶ҢмһҘн•ҳлҠ” мҪ”л“ң

```cs
public static void ClearAll<T>(this List<T> list)

    list.Capacity = 0;
    // лҳҗлҠ” лӢЁмҲңн•ҳкІҢ list = null; м°ёмЎ°к°Җ м—ҶлӢӨкі  м•ҢлҰј
}
```

мң„м—җм„ңмІҳлҹј `Clear()`лҘј мӮ¬мҡ©н•ҳм—¬ лҰ¬мҠӨнҠёк°Җ м°Ём§Җн•ҳкі  мһҲлҚҳ л©”лӘЁлҰ¬ кіөк°„мқ„ к°Җл№„м§Җ м»¬л үм…ҳмқҳ лҢҖмғҒмқҙ лҗҳкІҢ н•ңлӢӨ. к·ёлҰ¬кі  `Capacity = 0`мқ„ м Ғмҡ©н•ҳм—¬ лҰ¬мҠӨнҠёк°Җ лӮҙл¶Җм ҒмңјлЎң к°Җм§Җкі  мһҲлҠ” л°°м—ҙмқҳ л©”лӘЁлҰ¬лҸ„ к°Җл№„м§Җ м»¬л үм…ҳмқҳ лҢҖмғҒмқҙ лҗҳкІҢ н•ңлӢӨ. лӢ·л„·м—җм„ңлҠ” мқҙл ҮкІҢ н•ҳм—¬ мһҗлҸҷмңјлЎң GCк°Җ мқјм–ҙлӮҳлҸ„лЎқ нһҢнҠёлҘј мӨ„ мҲҳ мһҲлӢӨ. лӮҳлЁём§ҖлҠ” .NETмқҙ м•Ңм•„м„ң мһҗлҸҷ мІҳлҰ¬н•ңлӢӨ.

###### нҷ•мһҘ н•ЁмҲҳ кө¬нҳ„

Disposableн•ң м»¬л үм…ҳмқҖ `DisposeAll()`мқ„ мӮ¬мҡ©н•ҳкі  к·ёл Үм§Җ м•ҠмқҖ к°ңмІҙлҠ” `ClearAll()`мқ„ мӮ¬мҡ©н•ңлӢӨ.

```cs
public static class DisposableExtensions  
{  
    public static void DisposeAll<T>(this List<T> list) where T : IDisposable  
    {  
        list.ForIn(x => x.Dispose());  
    }  
  
    public static void DisposeAll<T>(this T[] array) where T : IDisposable  
    {  
        array.ForIn(x => x.Dispose());  
    }  
  
    public static void DisposeAll<T>(this IEnumerable<T> collection) where T : IDisposable  
    {  
        collection.ForIn(x => x.Dispose());  
    }  
  
    public static void ClearAll<T>(this List<T> list)  
    {  
        list.Clear();  
        list.Capacity = 0;  
    }  
  
    private static void ForIn<T>(this IEnumerable<T>? seq, Action<T> act) where T : IDisposable  
    {  
        if (seq == null)  
        {  
            return;  
        }  
  
        List<Exception> exceptions = [];  
  
        foreach (T item in seq)  
        {  
            try  
            {  
                act(item);  
            }  
            catch (Exception ex)  
            {  
                exceptions.Add(ex);  
            }  
        }  
  
        if (exceptions.Count > 0)  
        {  
            throw new AggregateException($"Disposing Item ERROR : {exceptions}");  
        }  
    }  
}
```

###### IDisposable нҢЁн„ҙ м Ғмҡ©

```cs
public sealed class DisposeList<T> : List<T>, IDisposable  
{  
    private bool mDisposedValue;  
  
    private void Dispose(bool disposing)  
    {  
        if (mDisposedValue)  
        {  
            return;  
        }  
  
        if (disposing)  
        {  
            foreach (IDisposable item in this.OfType<IDisposable>())  
            {  
                try  
                {  
                    item.Dispose();  
                }  
                catch (Exception ex)  
                {  
                    LogHelper.Logger.Error($"Disposing Item ERROR : {ex.Message}");  
                }  
            }  
  
            Clear();  
            Capacity = 0;  
        }  
  
        mDisposedValue = true;  
    }  
  
    ~DisposeList() => Dispose(false);  
  
    public void Dispose()  
    {  
        Dispose(disposing: true);  
        GC.SuppressFinalize(this);  
    }  
}
```

```cs
public sealed class DisposeArray<T>(uint initialLength = 0) : IDisposable  
{  
    private T[]? mArray = initialLength > 0 ? new T[initialLength] : null;  
    private bool mDisposedValue;  
  
    public T this[int index]  
    {  
        get  
        {  
            ObjectDisposedException.ThrowIf(mDisposedValue, nameof(DisposeArray<T>));  
  
            if (mArray == null || index < 0 || index >= Count)  
            {  
                throw new IndexOutOfRangeException();  
            }  
  
            return mArray[index];  
        }  
        set  
        {  
            ObjectDisposedException.ThrowIf(mDisposedValue, nameof(DisposeArray<T>));  
  
            if (index < 0)  
            {  
                throw new IndexOutOfRangeException();  
            }  
  
            if (mArray == null)  
            {  
                mArray = new T[Math.Max(index + 1, 4)];  
            }  
            else if (index >= mArray.Length)  
            {  
                Array.Resize(ref mArray, Math.Max(index + 1, mArray.Length * 2));  
            }  
  
            mArray[index] = value;  
            Count = Math.Max(Count, index + 1);  
        }  
    }  
  
    public int Count { get; private set; }  
  
    public int Capacity  
    {  
        get => mArray?.Length ?? 0;  
    }  
  
    public T[] ToArray()  
    {  
        ObjectDisposedException.ThrowIf(mDisposedValue, nameof(DisposeArray<T>));  
  
        if (mArray == null || Count == 0)  
        {  
            return [];  
        }  
  
        T[] newArray = new T[Count];  
        Array.Copy(mArray, newArray, Count);  
        return newArray;  
    }  
  
    public void Clear()  
    {  
        ObjectDisposedException.ThrowIf(mDisposedValue, nameof(DisposeArray<T>));  
  
        if (mArray != null)  
        {  
            Array.Clear(mArray, 0, mArray.Length);  
        }  
  
        Count = 0;  
    }  
  
    private void Dispose(bool disposing)  
    {  
        if (mDisposedValue) return;  
  
        if (disposing)  
        {  
            if (typeof(IDisposable).IsAssignableFrom(typeof(T)) && mArray != null)  
            {  
                List<Exception> exceptions = [];  
  
                foreach (IDisposable item in mArray.OfType<IDisposable>())  
                {  
                    try  
                    {  
                        item.Dispose();  
                    }  
                    catch (Exception ex)  
                    {  
                        exceptions.Add(ex);  
                    }  
                }  
  
                if (exceptions.Count != 0)  
                {  
                    throw new AggregateException($"Disposing Item ERROR : {exceptions}");  
                }  
            }  
            else  
            {  
                mArray = null;  
            }  
        }  
  
        mDisposedValue = true;  
    }  
  
    ~DisposeArray() => Dispose(false);  
  
    public void Dispose()  
    {  
        Dispose(true);  
        GC.SuppressFinalize(this);  
    }  
}
```

* м°ёкі лЎң мң„м—җм„ң м§Ғм ‘ мһ‘м„ұн•ң мҶҢмҠӨлҘј [**deepseek**](https://www.deepseek.com/)м—җ м§Ҳл¬ён–ҲлҚ”лӢҲ м•„лһҳмҷҖ к°ҷмқҖ лӢөліҖмқ„ н•ҙмЈјм—ҲлӢӨ. л№„көҗн•ҙм„ң н•ҷмҠөн•ҳлҠ” лҚ° лҸ„мӣҖмқҙ лҗҳкёё л°”лқјл©ҙм„ң мҶҢмҠӨмҪ”л“ңлҘј м•„лһҳм—җ мҳ®кІЁліёлӢӨ. мҳ®кё°лҠ” кіјм •м—җм„ң м•Ҫк°„мқҳ мҶҢмҠӨ м¶”к°Җ(`IEnumerable<T>`,`GetEnumerator` кҙҖл Ё)л°Ҹ м •лҰ¬лҠ” н•ҳмҳҖлӢӨ.

```cs
public sealed class DisposeList<T> : IDisposable, IEnumerable<T>  
{  
    private readonly List<T> mList = [];  
    private bool mDisposedValue;  
  
    public T this[int index]  
    {  
        get => mList[index];  
        set => mList[index] = value;  
    }  
  
    public int Count  
    {  
        get => mList.Count;  
    }  
  
    public int Capacity  
    {  
        get => mList.Capacity;  
    }  
  
    public void Add(T item) => mList.Add(item);  
  
    public bool Remove(T item) => mList.Remove(item);  
  
    public void Clear() => mList.Clear();

    public IEnumerator<T> GetEnumerator() => mList.GetEnumerator();

    IEnumerator IEnumerable.GetEnumerator() => GetEnumerator();
  
    private void Dispose(bool disposing)  
    {  
        if (mDisposedValue) return;  
  
        if (disposing)  
        {  
            List<Exception> exceptions = [];  
  
            foreach (T item in mList)  
            {  
                if (item is not IDisposable disposable)  
                {  
                    continue;  
                }  
  
                try  
                {  
                    disposable.Dispose();  
                }  
                catch (Exception ex)  
                {  
                    exceptions.Add(ex);  
                }  
            }  
  
            if (exceptions.Count > 0)  
            {  
                throw new AggregateException("DisposeList disposal failed", exceptions);  
            }  
  
            mList.Clear();  
        }  
  
        mDisposedValue = true;  
    }  
  
    ~DisposeList() => Dispose(false);  
  
    public void Dispose()  
    {  
        Dispose(true);  
        GC.SuppressFinalize(this);  
    }  
}
```

```cs
public sealed class DisposeArray<T> : IDisposable  
{  
    private T[]? mArray;  
    private bool mDisposedValue;  
  
    public DisposeArray(uint initialLength = 0)  
    {  
        if (initialLength > 0) mArray = new T[initialLength];  
    }  
  
    public T this[int index]  
    {  
        get  
        {  
            ObjectDisposedException.ThrowIf(mDisposedValue, this);  
  
            if (mArray == null || index < 0 || index >= Count)  
            {  
                throw new IndexOutOfRangeException();  
            }  
  
            return mArray[index];  
        }  
        set  
        {  
            ObjectDisposedException.ThrowIf(mDisposedValue, this);  
  
            if (index < 0)  
            {  
                throw new IndexOutOfRangeException();  
            }  
  
            if (mArray == null)  
            {  
                mArray = new T[Math.Max(index + 1, 4)];  
            }  
            else if (index >= mArray.Length)  
            {  
                Array.Resize(ref mArray, Math.Max(index + 1, mArray.Length * 2));  
            }  
  
            if (mArray[index] is IDisposable disposable)  
            {  
                disposable.Dispose();  
            }  
  
            mArray[index] = value;  
            Count = Math.Max(Count, index + 1);  
        }  
    }  
  
    public int Count { get; private set; }  
  
    public int Capacity  
    {  
        get => mArray?.Length ?? 0;  
    }  
  
    public T[] ToArray()  
    {  
        ObjectDisposedException.ThrowIf(mDisposedValue, this);  
        return mArray == null || Count == 0 ? [] : mArray[..Count];  
    }  
  
    private void Dispose(bool disposing)  
    {  
        if (mDisposedValue) return;  
  
        if (disposing && mArray != null)  
        {  
            List<Exception> exceptions = [];  
  
            foreach (T item in mArray)  
            {  
                if (item is not IDisposable disposable)  
                {  
                    continue;  
                }  
  
                try  
                {  
                    disposable.Dispose();  
                }  
                catch (Exception ex)  
                {  
                    exceptions.Add(ex);  
                }  
            }  
  
            if (exceptions.Count > 0)  
            {  
                throw new AggregateException(exceptions);  
            }  
  
            mArray = null;  
        }  
  
        mDisposedValue = true;  
    }  
  
    ~DisposeArray() => Dispose(false);  
  
    public void Dispose()  
    {  
        Dispose(true);  
        GC.SuppressFinalize(this);  
    }  
}
```
